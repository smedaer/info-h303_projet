DROP DATABASE IF EXISTS horeca;
CREATE DATABASE horeca CHARACTER SET 'utf8';
use horeca;

CREATE TABLE Etablissement (
	SID		INTEGER UNSIGNED UNIQUE AUTO_INCREMENT,
	Name		VARCHAR(30) NOT NULL,
	AdRue		VARCHAR(50),
	AdNumero	INTEGER UNSIGNED,
	AdCodePostal	INTEGER UNSIGNED,
	Longitude	INTEGER UNSIGNED,
	Latitude	INTEGER UNSIGNED,
	Tel		INTEGER UNSIGNED,
	PRIMARY KEY (SID)
);
